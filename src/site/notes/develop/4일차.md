        form = SignUpForm(request.POST)

@login_required(login_url='base:login')
def post_detail(request, pk):
    post = get_object_or_404(Post, pk=pk)
    comments = post.comments.all().order_by('created_at')
    
    if request.method == 'POST':
        comment_form = CommentForm(request.POST)
        if comment_form.is_valid():
            comment = comment_form.save(commit=False)
            comment.post = post
            comment.author = request.user
            comment.save()
            return redirect('base:post_detail', pk=post.pk)
    else:
        comment_form = CommentForm()

    return render(request, 'base/post_detail.html', {
        'post': post,
        'comments': comments,
        'comment_form': comment_form
    })

# 글 작성
@login_required(login_url='base:login')

            post.author = request.user
            post.save()
            return redirect('base:community')
    else:
        form = PostForm()
    return render(request, 'base/post_create.html', {'form': form})

from django import forms
from django.contrib.auth.models import User
from django.contrib.auth.forms import UserCreationForm
from .models import Post, Comment, Profile
from .models import Post, Comment, Profile, OCCUPATION_CHOICES

# 회원가입 폼 (직군 선택 추가)
class SignUpForm(UserCreationForm):
    # models.py에 있는 리스트(OCCUPATION_CHOICES)를 직접 가져다 쓰는 방식
    occupation = forms.ChoiceField(choices=OCCUPATION_CHOICES, label="직군")
    class Meta:
        model = User
        fields = ['username', 'email'] # 비밀번호는 UserCreationForm이 자동 처리

    def save(self, commit=True):
        user = super().save(commit=False)
        if commit:
            user.save()
            # 프로필에 직군 저장
            user.profile.occupation = self.cleaned_data['occupation']
            user.profile.save()
        return user

# 게시글 작성 폼
class PostForm(forms.ModelForm):
    class Meta:
        model = Post
        fields = ['title', 'content']
        widgets = {
            'title': forms.TextInput(attrs={'class': 'form-control', 'placeholder': '제목을 입력하세요'}),
            'content': forms.Textarea(attrs={'class': 'form-control', 'placeholder': '내용을 입력하세요', 'rows': 10}),
        }

# 댓글 작성 폼
class CommentForm(forms.ModelForm):
    class Meta:
        model = Comment
        fields = ['content']
        widgets = {
            'content': forms.Textarea(attrs={'class': 'form-control', 'placeholder': '댓글을 남겨주세요', 'rows': 3}),
        }

from django.contrib import admin
from django.contrib.auth.admin import UserAdmin as BaseUserAdmin
from django.contrib.auth.models import User
from .models import Post, Comment, Profile

# 1. 기존 User admin 설정 취소 (새로 덮어쓰기 위해)
admin.site.unregister(User)

# 2. 프로필을 유저 페이지 안에 껴넣기 (Inline 설정)
class ProfileInline(admin.StackedInline):
    model = Profile
    can_delete = False
    verbose_name_plural = '프로필 (직군 설정)'

# 3. 새로운 User admin 정의
class UserAdmin(BaseUserAdmin):
    inlines = (ProfileInline,)

# 4. 모델 등록
admin.site.register(User, UserAdmin)      # 유저 (프로필 포함)
admin.site.register(Post)                 # 게시글 관리
admin.site.register(Comment)              # 댓글 관리

#### [모델 적용]
아래 내용을 차례로 터미널에 입력합니다
1.DB 주소 연결
```
$env:DATABASE_URL="(전에 만들었던 님들 DB주소 입력)"
```
2.혹시 몰라서 하는거
```
python manage.py makemigrations
```
3.진짜 적용
```
python manage.py migrate
```
4.관리자 계정 다시 추가
```
python manage.py createsuperuser
```
5.깃에 올리기
```
git add .
git commit -m "Fix: DB Migration"
git push origin main
```
