계정 2단계 보안인증 하고 앱 비밀번호 만드시면 됩니다.

#### [백엔드 처리 & 라우팅]
`views.py` 하단에 추가합니다. 원래 정의는 상단에 추가하긴 하는데 귀찮음.
```python
# base/views.py

from django.shortcuts import render, redirect
from django.core.mail import send_mail
from django.conf import settings
from django.contrib import messages

def contact_view(request):
    if request.method == 'POST':
        # 1. 폼 데이터 추출
        name = request.POST.get('name')
        sender_email = request.POST.get('email')
        message = request.POST.get('message')

        # 2. 이메일 제목 및 본문 구성
        subject = f"[VERTRA 문의 접수] {name}님으로부터 새로운 메시지가 도착했습니다."
        full_message = f"발신자 이름: {name}\n발신자 이메일: {sender_email}\n\n[메시지 내용]\n{message}"

        try:
            # 3. 이메일 전송 실행
            send_mail(
                subject,
                full_message,
                settings.EMAIL_HOST_USER, # 발신자 (서버 계정)
                ['vertra.contact@gmail.com'], # 수신자 (받을 계정 리스트)
                fail_silently=False,
            )
            # 4. 성공 시 메시지 전달 및 새로고침
            messages.success(request, '성공적으로 메시지가 전송되었습니다. 담당자가 확인 후 회신드리겠습니다.')
            return redirect('base:contact')

        except Exception as e:
            # 실패 시 에러 처리
            messages.error(request, f'서버 오류로 전송에 실패했습니다: {e}')
            return redirect('base:contact')

    # GET 요청 시 단순 페이지 렌더링
    return render(request, 'base/contact.html')
```
